private void InsertionSort(ref int[] array)
{
    for (int i = 1; i < array.Length; i++)
    {
        int key = array[i],
            j = i - 1;

        while (j > -1 && array[j] > key)
        {
            array[j + 1] = key; //swap
            j -= 1;
        }

        array[j + 1] = key;
    }
}
